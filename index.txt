                                    if (context.parsed !== null) {

            function updateBudget() {
                const total = parseInt(totalBudgetSlider.value);
                const groomPercent = parseInt(groomBudgetSlider.value) / 100;

                totalBudgetValue.textContent = total;

                const groomAmount = total * groomPercent;
                const brideAmount = total - groomAmount;

                brideValue.textContent = brideAmount.toFixed(2);
                groomValue.textContent = groomAmount.toFixed(2);

                budgetChart.data.datasets[0].data[0] = brideAmount;
                budgetChart.data.datasets[0].data[1] = groomAmount;
